# Макс. длина цепочки из определенных символов
```python
# Цепочка из A, B, C, D, E, F. Макс длина цепочки из A, C, D, F
# Идея: если символ в ACDF, то значение в m +1 от предыдущего, иначе 0

# s BAAAAAQCADCADZZAA
# m 01234501234560012

s = open('24.txt').readline()
m = [0]*len(s)

for i in range(len(s)):
	if s[i] in 'ACDF':
		m[i] = m[i-1]+1
	# else: # <- можно не писать
	# m[i] = 0 # <- тк. стркоа 8
print(max(m))
```

# Подстрока, где нет сочетания символов
```python
# Макс кол-во идущих символов, среди которых нет сочетания РРР
# ААААРРРвВВВВРРРСССС
#  234562345678923456
m = [2]*len(s)
# Начинаем с 2, так как точно 2-ка символов точно подойдет
for i in range(2, len(s)):
	if s[i-2]+s[i-1]+s[i]!='PPP':

```
# Кол-во идущих подряд пар
```python
# Цепочка из X,Y,Z. Макс. кол-во идущих подряд пар символов ZX или ZY
# AZXZXZYZXBBBZXZY
# 0010203040000102
m = [0]*len(s)

for i in range(1, len(s)):
	if s[i-1]+s[i]=='ZX' or s[i-1]+s[i]=='ZY':
		# берем предпредыдущий, тк там счетчик пары
		m[i] = m[i-2]+1

print(max(m))
```
